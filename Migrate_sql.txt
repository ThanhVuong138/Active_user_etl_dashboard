SELECT A.ASSET, A.BUY_START_DT, B.STORE_ID, ORG.STORE_ID
FROM S_ASSET   A
     S_SRV_REQ SR
     S_ORDER   0
     S_ORG_EXT B
     S_ORG_EXT ORG
WHERE SR.ASSET_ID = A.ROW_ID
AND   O.BU_ID = B.ROW_ID
AND   O.SR_ID = SR.ROW_ID
AND   O.STATUS_CD = 'CLOSED'
AND   O.ORDER_NUM LIKE 'JC%'
AND   ORG.INTERGRATION_ID = B.INTERGRATION_ID
AND   SR.SR_CAT_TYPE_CD IN ('PI SERVICE') 
AND   SR.SERVICE_DATE > '01-JAN-2022'
AND   SR.SERVICE_DATE < '01-JAN-2023'
